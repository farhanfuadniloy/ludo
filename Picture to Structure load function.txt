/*
LOADING FUNCTION.
*/

void load(char *filename,struct pic *aa){
	FILE *fp = fopen(filename,"r");
	int xx,yy,temp;
	fscanf(fp,"%d %d",&xx,&yy);
	aa->x = xx, aa->y = yy;
	for(int i = 0; i<xx;i++){
		for(int j = 0; j<yy;j++){
			if(!feof(fp)){
				fscanf(fp,"%d",&temp);
				aa->pxl[i][j] = temp;
			}
		}
	}
	fclose(fp);
}
